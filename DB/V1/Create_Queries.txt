CREATE DATABASE FLIGHT_BOOKING;

CREATE TABLE user_account (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,

    username VARCHAR(255) NOT NULL,
    password VARCHAR(255) NOT NULL,

    first_name VARCHAR(255),
    last_name VARCHAR(255),

    phone_number VARCHAR(20),

    created_by VARCHAR(255) NOT NULL,
    created_at TIMESTAMP NOT NULL,

    modified_by VARCHAR(255) NOT NULL,
    modified_at TIMESTAMP NOT NULL,

    CONSTRAINT uk_user_account_username UNIQUE (username)
);


CREATE TABLE COUNTRY (
  id BIGINT AUTO_INCREMENT PRIMARY KEY,
  
  country_code VARCHAR(10) NOT NULL,
  country_name VARCHAR(255) NOT NULL,
  
  created_by VARCHAR(255) NOT NULL,
  created_at TIMESTAMP NOT NULL,

  modified_by VARCHAR(255) NOT NULL,
  modified_at TIMESTAMP NOT NULL,
 
  CONSTRAINT uk_country_name UNIQUE (country_name),
  CONSTRAINT uk_country_code UNIQUE (country_code)
);

CREATE TABLE CITY (
  id BIGINT AUTO_INCREMENT PRIMARY KEY,
  
  city_code VARCHAR(10) NOT NULL,
  city_name VARCHAR(255) NOT NULL,
  
  country_id BIGINT NOT NULL, 
  
  created_by VARCHAR(255) NOT NULL,
  created_at TIMESTAMP NOT NULL,

  modified_by VARCHAR(255) NOT NULL,
  modified_at TIMESTAMP NOT NULL,

  CONSTRAINT uk_city_code UNIQUE (city_code),
  CONSTRAiNT fk_country_id FOREIGN KEY (country_id) REFERENCES COUNTRY (id) 
);

CREATE TABLE AIRPORT (
   id BIGINT AUTO_INCREMENT PRIMARY KEY,
   
   airport_name VARCHAR(255) NOT NULL,
   city_id BIGINT NOT NULL, 
   
   created_by VARCHAR(255) NOT NULL,
   created_at TIMESTAMP NOT NULL,

   modified_by VARCHAR(255) NOT NULL,
   modified_at TIMESTAMP NOT NULL,
   
   CONSTRAINT uk_airport_name UNIQUE (airport_name),
   CONSTRAINT fk_city_id FOREIGN KEY (city_id) references City (id)
);


CREATE TABLE AIRLINE (
  id BIGINT AUTO_INCREMENT PRIMARY KEY,
  
  airline_name VARCHAR(255) NOT NULL,
  
  created_by VARCHAR(255) NOT NULL,
  created_at TIMESTAMP NOT NULL,

  modified_by VARCHAR(255) NOT NULL,
  modified_at TIMESTAMP NOT NULL,
  
  CONSTRAINT uk_airline_name UNIQUE (airline_name)
);


CREATE TABLE SEAT_TYPE (
 id BIGINT AUTO_INCREMENT PRIMARY KEY,
 
 type VARCHAR(255) NOT NULL,
 
 created_by VARCHAR(255) NOT NULL,
 created_at TIMESTAMP NOT NULL,

 modified_by VARCHAR(255) NOT NULL,
 modified_at TIMESTAMP NOT NULL,
  
 CONSTRAINT uk_type UNIQUE(type)
 );
 
 
CREATE TABLE FLIGHT  (
id BIGINT AUTO_INCREMENT PRIMARY KEY,
flight_number VARCHAR(255) NOT NULL,
airline_id BIGINT NOT NULL,

created_by VARCHAR(255) NOT NULL,
created_at TIMESTAMP NOT NULL,

modified_by VARCHAR(255) NOT NULL,
modified_at TIMESTAMP NOT NULL,

CONSTRAINT uk_flight_number UNIQUE(flight_number),
CONSTRAINT fk_flight_airline FOREIGN KEY (airline_id) REFERENCES airline (id)
);


CREATE TABLE FLIGHT_SEAT_CAPACITY (
   id BIGINT AUTO_INCREMENT PRIMARY KEY,
   
   flight_id BIGINT NOT NULL,
   seat_type_id BIGINT NOT NULL,
   capacity INT NOT NULL,
   
   created_by VARCHAR(255) NOT NULL,
   created_at TIMESTAMP NOT NULL,
   
   modified_by VARCHAR(255) NOT NULL,
   modified_at TIMESTAMP NOT NULL,
   
   CONSTRAINT fk_seat_cap_flight_id FOREIGN KEY (flight_id)  REFERENCES flight (id),
   CONSTRAINT fk_seat_cap_seat_type_id FOREIGN KEY (seat_type_id) REFERENCES seat_type (id)
   );
   
   
  CREATE TABLE flight_schedule (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,

    flight_id BIGINT NOT NULL,

    source_airport_id BIGINT NOT NULL,
    destination_airport_id BIGINT NOT NULL,

    departure_time TIMESTAMP NOT NULL,
    arrival_time TIMESTAMP NOT NULL,

    effective_from DATE NOT NULL,
    effective_to DATE NOT NULL,

    created_by VARCHAR(255) NOT NULL,
    created_at TIMESTAMP NOT NULL,

    modified_by VARCHAR(255) NOT NULL,
    modified_at TIMESTAMP NOT NULL,

    CONSTRAINT fk_flight_schedule_flight
        FOREIGN KEY (flight_id)
        REFERENCES flight (id),

    CONSTRAINT fk_flight_schedule_source_airport
        FOREIGN KEY (source_airport_id)
        REFERENCES airport (id),

    CONSTRAINT fk_flight_schedule_destination_airport
        FOREIGN KEY (destination_airport_id)
        REFERENCES airport (id),

    CONSTRAINT uk_flight_schedule_unique
        UNIQUE (flight_id, source_airport_id, destination_airport_id, effective_from)
);

CREATE TABLE booking (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,

    booking_reference VARCHAR(20) NOT NULL,
    user_id BIGINT NOT NULL,

    flight_schedule_id BIGINT NOT NULL,
    seat_type_id BIGINT NOT NULL,

    booking_status ENUM('HOLD', 'CONFIRMED', 'CANCELLED', 'FAILED') NOT NULL,
    travel_date DATE NOT NULL,

    price DECIMAL(10,2) NOT NULL,

    created_by VARCHAR(255) NOT NULL,
    created_at TIMESTAMP NOT NULL,

    modified_by VARCHAR(255) NOT NULL,
    modified_at TIMESTAMP NOT NULL,

    CONSTRAINT uk_booking_reference UNIQUE (booking_reference),

    CONSTRAINT fk_booking_user
        FOREIGN KEY (user_id) REFERENCES user_account (id),

    CONSTRAINT fk_booking_flight_schedule
        FOREIGN KEY (flight_schedule_id) REFERENCES flight_schedule (id),

    CONSTRAINT fk_booking_seat_type
        FOREIGN KEY (seat_type_id) REFERENCES seat_type (id)
);


CREATE TABLE booking_passenger (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,

    booking_id BIGINT NOT NULL,

    first_name VARCHAR(255) NOT NULL,
    last_name VARCHAR(255) NOT NULL,
    age INT NOT NULL,
    gender ENUM('MALE', 'FEMALE', 'OTHER') NOT NULL,

    CONSTRAINT fk_passenger_booking
        FOREIGN KEY (booking_id) REFERENCES booking (id)
);
